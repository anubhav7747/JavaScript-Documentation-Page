Problem 39. Create a glowing bulb effect using classlist toggle method in JavaScript.

<body>
    <div id="bulb" class="bulb-color bulb"></div>
</body>
<script>
    let bulb = () => {
        document.querySelector('#bulb').classList.toggle('bulb-color');
    }
                      
    setInterval(bulb, 500);
</script>
