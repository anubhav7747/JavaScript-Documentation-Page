Problem 7. Demonstrate the use of switch case statements in JavaScript.

const fruit = "apple";

switch(fruit) {
    case "apple" :
        console.log("Price 100Rs. per kg");     // this will print
        break;
    case "banana" :
        console.log("Price 40Rs per dozen");
        break;
    case "guava" :
        console.log("Price 50Rs per kg");
        break;
    default:
        console.log("Mixed fruit at Rs.60");
        break;
}
